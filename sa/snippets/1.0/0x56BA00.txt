float fromCoord[3], toCoord[3]

toCoord[0] = 1015.0
toCoord[1] = 220.13

// outputs
float out_CollisionPoint[3]
int out_CollidedEntity

int toCoordPtr = get_var_pointer toCoord
int collisionPointPtr = get_var_pointer out_CollisionPoint
int collidedEntityPtr = get_var_pointer out_CollidedEntity

int isCollidingSomething = CWorld_ProcessLineOfSight(fromCoordPtr, toCoordPtr, collisionPointPtr, collidedEntityPtr, isBuilding, isVehicle, isPed, isObject, isParticle, ignoreSeeThroughStuff, ignoreSomeObjectsForCamera, ignoreShootThroughStuff)
if isCollidingSomething == true
then print_formatted_now {format} "Obstacle detected between two points:~n~Entity: %p~n~%f %f %f" {time} 1000 {args} out_CollidedEntity ...out_CollisionPoint
end
